// ===== Simple Catches UI (frontend) =====

// Auto-detect backend or override with ?api=http://host:port
const qp = new URLSearchParams(location.search);
const DEFAULT_API = `${location.protocol}//${location.hostname}:4000`;
const API = qp.get('api') || DEFAULT_API; // e.g. http://127.0.0.1:4000
console.log('API base:', API);

// DOM
const apiStatus  = document.getElementById('apiStatus');
const tbody      = document.getElementById('tbody');
const addBtn     = document.getElementById('addBtn');
const refreshBtn = document.getElementById('refreshBtn');

const email     = document.getElementById('email');
const species   = document.getElementById('species');
const weight_kg = document.getElementById('weight_kg');
const length_cm = document.getElementById('length_cm');
const lat       = document.getElementById('lat');
const lng       = document.getElementById('lng');
const notes     = document.getElementById('notes');

// ---- helpers
function fmtDate(iso) {
  if (!iso) return '';
  const d = new Date(iso);
  return isNaN(d.getTime()) ? String(iso) : d.toLocaleString();
}

function fmtNum(n, dp = 4) {
  const x = Number(n);
  return Number.isFinite(x) ? x.toFixed(dp) : '';
}

async function jsonFetch(url, opts = {}) {
  const ctrl = new AbortController();
  const timer = setTimeout(() => ctrl.abort(), 15000);
  try {
    const res = await fetch(url, { ...opts, signal: ctrl.signal });
    let data = null;
    try { data = await res.json(); } catch {}
    if (!res.ok) {
      const msg = (data && (data.error || data.message)) || `HTTP ${res.status}`;
      throw new Error(msg);
    }
    return data;
  } finally {
    clearTimeout(timer);
  }
}

async function checkApi() {
  try {
    const j = await jsonFetch(`${API}/health`);
    if (j?.ok) {
      apiStatus.textContent = 'API OK';
      apiStatus.className = 'ok';
    } else {
      throw new Error('Health not ok');
    }
  } catch (e) {
    apiStatus.textContent = 'API not reachable';
    apiStatus.className = 'error';
    console.warn('Health check failed:', e.message || e);
  }
}

function rowEl(c) {
  const tr = document.createElement('tr');

  const tdWhen = document.createElement('td');
  tdWhen.className = 'nowrap';
  tdWhen.textContent = fmtDate(c.created_at);
  tr.appendChild(tdWhen);

  const tdEmail = document.createElement('td');
  tdEmail.textContent = c.user_email || '';
  tr.appendChild(tdEmail);

  const tdSpecies = document.createElement('td');
  tdSpecies.textContent = c.species || '';
  tr.appendChild(tdSpecies);

  const tdW = document.createElement('td');
  tdW.textContent = (c.weight_kg ?? '') === '' ? '' : String(c.weight_kg);
  tr.appendChild(tdW);

  const tdL = document.createElement('td');
  tdL.textContent = (c.length_cm ?? '') === '' ? '' : String(c.length_cm);
  tr.appendChild(tdL);

  const tdLL = document.createElement('td');
  tdLL.className = 'latlng';
  tdLL.textContent =
    (c.lat != null && c.lng != null)
      ? `${fmtNum(c.lat)}, ${fmtNum(c.lng)}`
      : '';
  tr.appendChild(tdLL);

  const tdNotes = document.createElement('td');
  const input = document.createElement('input');
  input.style.width = '100%';
  input.value = c.notes ?? '';
  input.setAttribute('data-notes-for', c.id);
  tdNotes.appendChild(input);
  tr.appendChild(tdNotes);

  const tdActions = document.createElement('td');
  tdActions.className = 'actions';
  const btnSave = document.createElement('button');
  btnSave.textContent = 'Save';
  const btnDel = document.createElement('button');
  btnDel.textContent = 'Delete';
  tdActions.appendChild(btnSave);
  tdActions.appendChild(btnDel);
  tr.appendChild(tdActions);

  // wire buttons
  btnSave.onclick = async () => {
    try {
      const body = { notes: input.value };
      await jsonFetch(`${API}/catches/${c.id}`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(body),
      });
      btnSave.textContent = 'Saved';
      setTimeout(() => (btnSave.textContent = 'Save'), 800);
    } catch (e) {
      alert('Save failed: ' + (e.message || e));
    }
  };

  btnDel.onclick = async () => {
    if (!confirm('Delete this catch?')) return;
    try {
      await jsonFetch(`${API}/catches/${c.id}`, { method: 'DELETE' });
      await loadRows();
    } catch (e) {
      alert('Delete failed: ' + (e.message || e));
    }
  };

  return tr;
}

async function loadRows() {
  tbody.innerHTML = '';
  try {
    const j = await jsonFetch(`${API}/catches`);
    // Support both shapes: array OR { success, data }
    const list = Array.isArray(j) ? j : (Array.isArray(j?.data) ? j.data : []);
    if (!list.length) {
      tbody.innerHTML = `<tr><td colspan="8">No data</td></tr>`;
      return;
    }
    const frag = document.createDocumentFragment();
    list.forEach((c) => frag.appendChild(rowEl(c)));
    tbody.appendChild(frag);
  } catch (e) {
    console.error('loadRows failed:', e);
    tbody.innerHTML = `<tr><td colspan="8">Failed to load: ${e.message || e}</td></tr>`;
  }
}

addBtn.onclick = async () => {
  const body = {
    user_email: email.value.trim(),
    species:    species.value.trim(),
    weight_kg:  Number(weight_kg.value || 0) || null,
    length_cm:  Number(length_cm.value || 0) || null,
    lat:        lat.value ? Number(lat.value) : null,
    lng:        lng.value ? Number(lng.value) : null,
    notes:      notes.value || null,
  };

  // basic validation
  if (!body.user_email || !body.species) {
    alert('email and species are required');
    return;
  }

  try {
    await jsonFetch(`${API}/catches`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    });

    // quick refresh + clear inputs
    email.value = '';
    species.value = '';
    weight_kg.value = '';
    length_cm.value = '';
    lat.value = '';
    lng.value = '';
    notes.value = '';
    await loadRows();
  } catch (e) {
    alert('Add failed: ' + (e.message || e));
  }
};

refreshBtn.onclick = loadRows;

// initial
await checkApi();
await loadRows();
